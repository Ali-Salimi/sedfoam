95  //#include "readPPProperties.H"
96  #include "initContinuityErrs.H"
97  #include "createTimeControls.H"

99  #include "setInitialDeltaT.H"
100  // pimpleControl pimple(mesh);
101 
102  // * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * * //
103  if (SUSlocal)
104  {
105  Info<< "\nLocal Schmidt number activated" << endl;
106  }
107  else
108  {
109  if (max(SUS).value() == 0)
110  {
111  Info<< "Turbulence suspension term is neglected" << endl;
112  }
113  else if (max(SUS).value() > 0)
114  {
115  Info<< "Turbulence suspension term is included" << endl;
116  }
117  else
118  {
119  Info<< "Turbulence suspension coefficient SUS can't be negative"
120  << endl;
121  }
122  }
123 
124  Info<< "\nStarting time loop\n" << endl;
125 
126  while (runTime.run())
127  {
128  #include "readTwoPhaseEulerFoamControls.H"
129  #include "CourantNos.H"
130  #include "alphaCourantNo.H"
131  #include "setDeltaT.H"
132 
133  runTime++;
134  Info<< "Time = " << runTime.timeName() << nl << endl;
135 
136  #include "gravityRamp.H"

138 // Pressure-velocity PIMPLE corrector loop
139  while (pimple.loop())
140  {
141  #include "alphaEqn.H"
142  #include "liftDragCoeffs.H"
143 
144 // Compute the Kinetic Theory parameters: nuEffa and lambdaUa from the

146  #include "callKineticTheory.H"
147 
148 // Compute the contact pressure pff and the Frictional stress nuFra
149 // from a Coulomb model if using the kinetic theory
150 // and from the mu(I) rheology if using the granular rheology
151  #include "callFrictionStress.H"
152 
153 // Create the momentum balance equations for both phases a and b
154  #include "UEqns.H"
155 
156  #include "pEqn.H"
157 
158  #include "DDtU.H"
159 
160  if (pimple.turbCorr())
161  {
162  #include "updateTwoPhaseRASTurbulence.H"
163  turbulenceb->correct();
164  if (debugInfo)
165  {
166  Info << " max(nutb) = "
167  << max(turbulenceb->nut()).value() << endl;
168  }
169  }
170  }
171  if (debugInfo)
172  {
173  Info<< "min(Ua) = " << gMin(Ua)
174  << "max(Ua) = " << gMax(Ua) << endl;
175  Info<< "min(Ub) = " << gMin(Ub)
176  << "max(Ub) = " << gMax(Ub) << nl << endl;
177  }
178  #include "OutputGradPOSC.H"
179  #include "writeTau.H"
180 
181  Info<< "ExecutionTime = " << runTime.elapsedCpuTime() << " s"

186  Info<< "End\n" << endl;
187 
188  return 0;
189 }
